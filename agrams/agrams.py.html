  11 def getwords(astring):
  12   stringasset=set(astring)
  13   allwords=getwordlist()

  15   for word in allwords:
  16     if (set(word).issubset(stringasset)&checkletters(astring,word)):
  17       words.append(word)
  18   return words

  20 def getwordlist():
  21   words=set()
  22   try:
  23     wordfile=open('mypy/words.txt','r')
  24   except IOError:
  25     print "Could not open the word file. Please make sure that it is available"
  26   for word in wordfile:
  27     words.add(word[:-1])
  28   wordfile.close()
  29   return words

  31 def checkletters(mainstring, teststring):
